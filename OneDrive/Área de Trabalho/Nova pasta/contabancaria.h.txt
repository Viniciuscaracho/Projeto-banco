#include <iostream>
#include <cstring>

using namespace std;
class ContaBancaria {
    private:
        string nome;
        int cpf;
      

    public:
    	double saldo;
        ContaBancaria(string _nome, int _cpf, double _saldo) {
            nome = _nome;
            cpf = _cpf;
            saldo = _saldo;
        }

        void exibir_cont() {
            cout << "Nome do titular: " << nome << endl;
            cout << "CPF cadastrado: " << cpf << endl;
            cout << "Saldo atual: " << saldo << endl;
        }

        void depositar(double valor) {
            saldo += valor;
        }

        void saque(double valor) {
            if(valor <= saldo) {
                saldo -= valor;
            } else {
                cout << "Saldo insuficiente" << endl;
            }
        }

        void transacao(double valor, ContaBancaria& outraConta) {
            if(valor <= saldo) {
                saldo -= valor;
                outraConta.depositar(valor);
            } else {
                cout << "Saldo insuficiente" << endl;
            }
        }
};
